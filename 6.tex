            <input type="number" placeholder="Enter First Number" id="num1" required><br>
            <input type="number" placeholder="Enter First Number" id="num2" required><br>
            <select id="operation">
                <option value="+">Sum</option>
                <option value="-">Minus</option>
                <option value="*">Multiply</option>
                <option value="/">Divide</option>
                <option value="%">Remainder</option>
                <option value="**">Power</option>
            </select><br>
            <button>Calculate</button><br>

    <script>
        const form = document.querySelector('form');
        const container = document.querySelector('.table')
        form.addEventListener('submit', function (e) {
            e.preventDefault()
            const num1 = document.querySelector('#num1').value;
            const num2 = document.querySelector('#num2').value;
            const operation = document.querySelector('#operation').value
            const statement = `${num1} ${operation} ${num2}`
            const result = eval(statement);
            const resultdiv = document.createElement('button')
            resultdiv.innerHTML = result;
            resultdiv.classList.add('result')
            form.appendChild(resultdiv)
        })
    </script>
